## The algorithm
```javascript title="hadley-attractor.js"

const a = 0.2;
const b = 4;
const f = 8;
const g = 1;

const next = (x,y,z) => {
  const dx = -Math.pow(y, 2) - Math.pow(z, 2) - a * x + a * f;
  const dy = x * y - b * x * z - y + g;
  const dz = b * x * y + x * z - z;
  return {dx,dy,dz};
};
```
